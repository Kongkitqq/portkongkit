import pandas as pd
from sklearn import tree
from sklearn.tree import DecisionTreeClassifier
import matplotlib.pyplot as plt

#  Iris dataset
from sklearn.datasets import load_iris
iris = load_iris()

#  DataFrame 
df = pd.DataFrame(iris.data, columns=iris.feature_names)

#  features  target
features = ['petal width (cm)', 'petal length (cm)', 'sepal length (cm)']  # 
X = df[features]
y = iris.target  #  target  iris 

#  Decision Tree
dtree = DecisionTreeClassifier()
dtree = dtree.fit(X, y)

#  Decision Tree
plt.figure(figsize=(12,8))
tree.plot_tree(dtree, feature_names=features, class_names=iris.target_names, filled=True)
plt.show()
