import pandas as pd
from sklearn import tree
from sklearn.tree import DecisionTreeClassifier
import matplotlib.pyplot as plt

# โหลด Iris dataset
from sklearn.datasets import load_iris
iris = load_iris()

# สร้าง DataFrame และกำหนดชื่อคอลัมน์
df = pd.DataFrame(iris.data, columns=iris.feature_names)

# กำหนด features และ target
features = ['petal width (cm)', 'petal length (cm)', 'sepal length (cm)']  # ชื่อคอลัมน์ที่ถูกต้อง
X = df[features]
y = iris.target  # ใช้ target จาก iris โดยตรง

# สร้างและฝึกโมเดล Decision Tree
dtree = DecisionTreeClassifier()
dtree = dtree.fit(X, y)

# วาด Decision Tree
plt.figure(figsize=(12,8))
tree.plot_tree(dtree, feature_names=features, class_names=iris.target_names, filled=True)
plt.show()
